## Notebook Snippets

```python
# Difference-in-Differences model
df["treated"] = (df["ticker"] == "FFIV").astype(int)
df["post"] = (df["date"] >= event_date).astype(int)
df["treated_post"] = df["treated"] * df["post"]
model = sm.OLS(df["returns"], sm.add_constant(df[["treated", "post", "treated_post"]])).fit()
```

```python
# Placebo test (shift event date)
placebo_date = event_date - pd.Timedelta(days=7)
df["post_placebo"] = (df["date"] >= placebo_date).astype(int)
df["treated_post_placebo"] = df["treated"] * df["post_placebo"]
placebo = sm.OLS(df["returns"], sm.add_constant(df[["treated", "post_placebo", "treated_post_placebo"]])).fit()
```

```python
# Run Difference-in-Differences regression from the notebook
def run_diff_in_diff(df, event_date, treated):
    start_date = event_date - datetime.timedelta(days=180)
    df = df[(df["date"] >= start_date) & (df["date"] <= event_date)].copy()
    df["treated"] = np.where(df["ticker"] == treated, 1, 0)
    df["post"] = np.where(df["date"] >= event_date, 1, 0)
    df["treated_post"] = df["treated"] * df["post"]
    model = smf.ols("returns ~ treated + post + treated_post", data=df).fit()
    return model
```
